-- 뷰생성
CREATE or REPLACE view MemberVO AS 
 ((SELECT s.sid as id, 'student' as type FROM student s)
 UNION (SELECT p.pid, 'professor' as type FROM professor p))
 UNION (SELECT a.aid, 'administer' as type FROM administer a);

-- 뷰삭제
 drop view MemberVO;
